Основные сущности и их назначение:

Users - представляет зарегистрированных пользователей системы.
Categories – классификация статей по тематикам.
Articles – основной контент системы - новости и статьи.


Users:  id - уникальный идентификатор пользователя, 
        user_name - имя пользователя (уникальное), 
        email - Email адрес (уникальный), 
        password_hash - хеш пароля, 
        role - роль пользователя (user/admin), 
        created_at - дата регистрации, 
        is_active - активность аккаунта (1-активен, 0-заблокирован).

Categories: id - уникальный идентификатор категории,    
            name - название категории, 
            slug - URL-friendly название для ссылок, 
            description - описание категории, 
            created_at - дата создания категории.

Articles:   id - уникальный идентификатор статьи, 
            title - заголовок статьи, 
            slug - URL-friendly заголовок, 
            content - полный текст статьи, 
            excerpt - краткое описание/аннотация, 
            author_id - ID автора, 
            category_id - ID категории, 
            status - статус публикации, 
            featured_image - путь к изображению, 
            views_count - счетчик просмотров, 
            created_at - дата создания, 
            updated_at - дата последнего изменения. 




CREATE TABLE users (
    id INT AUTO_INCREMENT PRIMARY KEY,
    user_name VARCHAR(100) NOT NULL UNIQUE,
    email VARCHAR(100) NOT NULL UNIQUE,
    password_hash VARCHAR(255) NOT NULL,
    role ENUM('user', 'admin') DEFAULT 'user',
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    is_active TINYINT(1) DEFAULT 1
);


CREATE TABLE categories (
    id INT AUTO_INCREMENT PRIMARY KEY,
    name VARCHAR(255) NOT NULL,
    slug VARCHAR(255) NOT NULL,
    description TEXT,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);





CREATE TABLE articles (
    id INT AUTO_INCREMENT PRIMARY KEY,
    title VARCHAR(255) NOT NULL,
    slug VARCHAR(255) NOT NULL,
    content TEXT,
    excerpt TEXT,
    author_id INT NOT NULL,
    category_id  INT NOT NULL,
    status ENUM('draft', 'published', 'archived') DEFAULT 'draft',
    featured_image VARCHAR(500),
    views_count INT DEFAULT 0,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,


    FOREIGN KEY (author_id) REFERENCES users(id),
    FOREIGN KEY (category_id) REFERENCES categories(id)
);
